CLASS Pet
    ATTRIBUTES: name, age
    CONSTRUCTOR(name, age)
    METHOD displayInfo()

CLASS Dog EXTENDS Pet
    ATTRIBUTE: breed
    CONSTRUCTOR(name, age, breed)
    METHOD displayInfo() → call super + print breed

CLASS Cat EXTENDS Pet
    ATTRIBUTE: color
    CONSTRUCTOR(name, age, color)
    METHOD displayInfo() → call super + print color

CLASS Bird EXTENDS Pet
    ATTRIBUTE: canFly
    CONSTRUCTOR(name, age, canFly)
    METHOD displayInfo() → call super + print canFly

MAIN
    CREATE Dog("Bruno", 3, "Labrador")
    CREATE Cat("Kitty", 2, "White")
    CREATE Bird("Parrot", 1, true)
    CALL displayInfo() for each
